if __name__ == '__main__':
    x = int(input())
    y = int(input())
    z = int(input())
    n = int(input())
count=1
    
print("[",end="")
for i in range(0,x+1):
    for j in range(0,y+1):
        for k in range(0,z+1):
            if((i+j+k)==n):
                continue
            else:
                if(count!=1):
                    print(", ",end="")
                print("[{}, {}, {}]".format(i,j,k),end="")
                count+=1
                
print("]",end="")
